        function createCard(mod) {
            const card = document.createElement('div');
            card.className = `mod-card rounded-xl p-4 transition-all duration-500 ease-in-out`;
            card.setAttribute('data-game', mod.game || 'CSSO');
            card.setAttribute('data-name', (mod.search || mod.name).toLowerCase());
            card.setAttribute('data-model', mod.model || '');

            let tagClass = (mod.game === 'CSSO' ? 'game-tag-csso' : 'game-tag-cs2');
            if (mod.label === "WEAPON" || mod.cat === "Knife" || mod.cat === "Pistol") tagClass = "game-tag-weapon";
            if (mod.label === "GLOVE") tagClass = "game-tag-glove";

            card.innerHTML = `
                <div class="cs-icon-wrapper">
                    <img src="https://images.steamusercontent.com/ugc/5838428307091219404/3E98FE0DBDB41DA63E793DEF2E3233C034F113DF/?imw=512&imh=512&ima=fit&impolicy=Letterbox&imcolor=%23000000&letterbox=true" 
                         class="cs-corner-icon" alt="Icon">
                </div>
                <div class="relative mb-4 aspect-video bg-black rounded-lg overflow-hidden flex items-center justify-center">
                    <img src="${mod.img}" class="w-full h-full object-cover opacity-80">
                    <span class="absolute top-2 left-2 category-badge text-[10px] font-black px-2 py-0.5 rounded">${mod.cat}</span>
                    <span class="absolute bottom-2 right-2 text-[10px] font-black ${tagClass} uppercase">
                        ${mod.label || 'WEAPON'}
                    </span>
                </div>
                <h3 class="text-white font-bold text-lg mb-1">${mod.name}</h3>
                <p class="text-gray-500 text-xs mb-4 flex-grow">${mod.desc || 'Sin descripción disponible.'}</p>
                <div class="flex justify-between items-center mt-auto pt-4">
                    <span class="text-green-500 font-bold text-sm">Gratis</span>
                    <a href="${mod.link}" target="_blank" class="btn-download text-xs bg-white/5 border border-white/10 px-3 py-1 rounded text-center">Descargar</a>
                </div>
            `;
            return card;
        }

        function filterByModel(modelName, btnElement) {
            document.querySelectorAll('.sidebar-item').forEach(el => el.classList.remove('active'));
            btnElement.classList.add('active');

            const grid = document.getElementById('weaponGrid');
            const cards = grid.querySelectorAll('.mod-card');
            
            cards.forEach(card => {
                card.style.opacity = '0';
                card.style.transform = 'translateY(10px)';
            });

            setTimeout(() => {
                cards.forEach(card => {
                    const modModel = card.getAttribute('data-model');
                    const shouldShow = (modelName === "All Pistols" || modelName === "All Knives" || modModel === modelName);
                    
                    if (shouldShow) {
                        card.classList.remove('hidden-mod');
                        void card.offsetWidth;
                        card.style.opacity = '1';
                        card.style.transform = 'translateY(0)';
                    } else {
                        card.classList.add('hidden-mod');
                    }
                });
            }, 300);
        }

        function showSection(sectionKey) {
            const h = document.getElementById('mainHeader');
            const m = document.getElementById('mainContent');
            const w = document.getElementById('weaponSection');
            const data = sections[sectionKey];

            h.classList.remove('page-active');
            m.classList.remove('page-active');
            w.classList.remove('page-active');

            setTimeout(() => {
                h.classList.add('hidden');
                m.classList.add('hidden');
                w.classList.remove('hidden');
                
                document.getElementById('sectionHeaderTitle').innerText = data.title;
                document.getElementById('sectionBreadcrumb').innerText = data.breadcrumb;
                document.getElementById('sidebarTitle').innerText = data.sidebarTitle;
                
                const sidebarList = document.getElementById('sidebarList');
                sidebarList.innerHTML = '';
                data.sidebarItems.forEach((item, idx) => {
                    const btn = document.createElement('button');
                    btn.className = `sidebar-item w-full text-left px-4 py-3 rounded-lg block ${idx === 0 ? 'active' : ''}`;
                    btn.innerText = item;
                    btn.onclick = () => filterByModel(item, btn);
                    sidebarList.appendChild(btn);
                });

                const grid = document.getElementById('weaponGrid');
                grid.innerHTML = '';
                data.data.forEach(item => grid.appendChild(createCard(item)));
                
                setTimeout(() => {
                    w.classList.add('page-active');
                }, 50);
            }, 400);

            window.scrollTo({ top: 0, behavior: 'smooth' });
        }

        function showMain() {
            const h = document.getElementById('mainHeader');
            const m = document.getElementById('mainContent');
            const w = document.getElementById('weaponSection');

            w.classList.remove('page-active');

            setTimeout(() => {
                w.classList.add('hidden');
                h.classList.remove('hidden');
                m.classList.remove('hidden');
                
                renderMods();

                setTimeout(() => {
                    h.classList.add('page-active');
                    m.classList.add('page-active');
                }, 50);
            }, 400);

            window.scrollTo({ top: 0, behavior: 'smooth' });
        }

        function renderMods() {
            const grid = document.getElementById('modGrid');
            grid.innerHTML = '';
            modsData.forEach(mod => grid.appendChild(createCard(mod)));
        }

        const searchInput = document.getElementById('modSearch');
        searchInput.addEventListener('input', () => {
            const term = searchInput.value.toLowerCase();
            document.querySelectorAll('#modGrid .mod-card').forEach(mod => {
                const name = mod.getAttribute('data-name');
                mod.classList.toggle('hidden-mod', !name.includes(term));
            });
        });

        function filterGame(game) {
            showMain();
            setTimeout(() => {
                document.querySelectorAll('#modGrid .mod-card').forEach(mod => {
                    const modGame = mod.getAttribute('data-game');
                    mod.classList.toggle('hidden-mod', game !== 'all' && modGame !== game);
                });
            }, 450);
        }

        renderMods();
    </script>
